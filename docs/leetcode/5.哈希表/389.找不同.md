## 解题代码

```js
/**
 * @param {string} s
 * @param {string} t
 * @return {character}
 */
var findTheDifference = function(s, t) {
  if (!s.length) return t;
  let sum1 = 0, sum2 = 0, i = 0;

  while (i < t.length) {
    if (s[i]) sum1 += s.charCodeAt(i);
    if (t[i]) sum2 += t.charCodeAt(i);
    i++;
  }
  return String.fromCharCode(sum2 - sum1);
};
```
