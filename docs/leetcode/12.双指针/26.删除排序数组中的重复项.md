## 解题代码

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var removeDuplicates = function(nums) {
  let start = 0;
  if (!nums.length) return start;

  for(let i = 1; i < nums.length; i++) {
    if (nums[start] !== nums[i]) {
      start++;
      nums[start] = nums[i];
    }
  }

  return start + 1;
};
```
