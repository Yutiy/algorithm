## 解题代码

```js
/**
 * @param {number[]} numbers
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(numbers, target) {
  let i = 0
  let j = numbers.length - 1

  while(i < j) {
    const diff = target - (numbers[i] + numbers[j])
    if (diff == 0) {
      return [i+1, j+1]
    } else if (diff > 0) {
      i++
    } else {
      j--
    }
  }

  return [-1, -1]
};
```
