## 解题代码

```js
/**
 * @param {number} x
 * @return {number}
 */
var reverse = function(num) {
  const number = Math.pow(2, 31)
  const min = -number;
  const max = number - 1;

  let res = 0;
  while(num !== 0) {
    res = 10 * res + num % 10;
    if(res > max || res < min) return 0;
    num = Math.trunc(num / 10);   // 取整
  }

  return res;
};
```
