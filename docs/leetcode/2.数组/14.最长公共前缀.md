## 解题代码

```js
/**
 * @param {string[]} strs
 * @return {string}
 */
var longestCommonPrefix = function(strs) {
  const len = strs.length;
  if(len == 0) return "";
  if(len == 1) return strs[0];

  let i = 1;
  let prefix = strs[0];

  while(prefix && i < len) {
    if(strs[i].indexOf(prefix) !== 0) {
      prefix = prefix.slice(0, prefix.length - 1);
    } else {
      i++;
    }
  }
  return prefix;
};
```
