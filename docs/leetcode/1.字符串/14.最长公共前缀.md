## 解题代码

```js
/**
 * @param {string[]} strs
 * @return {string}
 */
var longestCommonPrefix = function(strs) {
  const len = strs.length;
  if(len == 0) return "";
  if(len == 1) return strs[0];

  let i = 1;
  let perfix = strs[0];

  while(perfix && i < len) {
    if(strs[i].indexOf(perfix) !== 0) {
      perfix = perfix.slice(0, perfix.length - 1);
    } else {
      i++;
    }
  }
  return perfix;
};
```
